# PROTOCOL

## Инициализация клиента

```
1. (sk, pk) <- G()          // генерация приватного / публичного ключей
2. p <- P(pk)               // подтверждение публичного ключа доказательством работы
3. (p, pk) -> SERVER        // регистрация публичного ключа на сервере
```

## Подгрузка клиентов

```
1. H(req_pk) -> SERVER      // запрос на получение публичного ключа по хешу
2. (p, pk) <- SERVER        // получение ключа с доказательством работы
3. ok ?= Vp(pk, p)          // проверка доказательства работы
4. H(req_pk) ?= H(pk)       // проверка полученного публичного ключа
5. pk -> STORAGE            // сохранение ключа
```

## Инициализация канала

```
1. k <- G()                                                 // ключ канала
2. enc_n <- E(k,name)                                       // зашифрованное название канала
2. PL = (H(my_pk), H(pk1), H(pk2), ..., H(pkN))             // хеши публичных ключей участников канала
3. EL = (E(my_pk, k), E(pk1, k), E(pk2, k), ..., E(pkN, k)) // зашифрованный ключ канала публичными ключами участников
4. id <- HMAC(k, (enc_n || PL || EL))                       // идентификатор канала
5. s <- S(my_sk, id)                                        // подпись канала
6. p <- P(id)                                               // подтверждение идентификатора канала доказательством работы
7. (id, enc_n, PL, EL, p, s) -> SERVER                      // отправление информации о канале серверу
```

## Подгрузка каналов

```
1. (id, enc_n, PL, EL, p, s) <- SERVER  // получение канала
2. ok ?= Vs(PL[0], id, s)               // проверка подписи
3. ok ?= Vp(id, p)                      // проверка доказательства работы
4. index <- H(my_pk) ?in PL             // проверка моего ключа в списке
5. k <- D(my_sk, EL[index])             // получение ключа канала
6. id ?= HMAC(k, (n || PL || EL))       // проверка идентификатора канала
7. n <- D(k, enc_n)                     // получение названия канала
8. (id, k, n, PL) -> STORAGE            // сохранение канала
```

## Отправление сообщения в канал

```
1. k <- STORAGE[id]                     // получение ключа канала 
2. c <- E(k, m)                         // создание шифртекста
3. h <- HMAC(id, c)                     // хеш шифртекста
4. s <- S(my_pk, h)                     // подпись хеша
5. p <- P(h)                            // подтверждение хеша доказательством работы
6. (id, H(my_pk), c, s, p) -> SERVER    // отправление сообщения на сервер 
```

## Получение сообщения из канала

```
1. (id, H(pk), c, s, p) <- SERVER   // получение сообщения от сервера
2. ok ?= Vs(pk, c, s)               // проверка подписи участника
3. ok ?= Vp(c, p)                   // проверка доказательства работы
4. (k, PL) <- STORAGE[id]           // получение ключа и списка участников канала 
5. H(pk) ?in PL                     // проверка принадлежности участника к каналу
6. m <- D(k, c)                     // расшифрование сообщения
7. m -> STORAGE                     // сохранение сообщения
```
