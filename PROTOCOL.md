# PROTOCOL

## Инициализация клиента
> /client/init

```
1. (sk, pk) <- G()          // генерация приватного / публичного ключей
2. p = P(pk)                // подтверждение публичного ключа доказательством работы
3. (p, pk) -> SERVER        // регистрация публичного ключа на сервере
```

## Подгрузка клиентов
> /client/load

```
1. (p, pk) <- SERVER        // получение ключа
2. ok =? V(pk, p)           // проверка доказательства работы
3. pk -> STORAGE            // сохранение ключа
```

## Инициализация канала

```
1. k <- G()                                                 // ключ канала
2. n <- E(k,name)                                           // называние канала
2. RL = (my_pk, pk1, pk2, ..., pkN)                         // участники канала
3. EL = (E(my_pk, k), E(pk1, k), E(pk2, k), ..., E(pkN, k)) // шифрование ключа канала
4. id <- HMAC(k, n || RL || EL)                             // идентификатор канала
5. s = S(sk, id)                                            // подпись канала
6. p = P(id)                                                // доказательство работы канала
7. (id, n, RL, EL, p, s) -> SERVER                          // отправление информации о канале серверу
```

## Подгрузка каналов

```
1. (id, RL, EL, p, s) <- SERVER // получение канала
2. index <- my_pk in? RL        // проверка моего ключа в списке
3. k <- D(sk, EL[index])        // получение ключа канала
4. id =? HMAC(k, RL || EL)      // проверка идентификатора канала
5. ok =? V(RL[0], id, s)        // проверка подписи
6. ok =? V(id, p)               // проверка доказательства работы
7. (id, k, RL) -> STORAGE       // сохранение канала
```

## Отправление сообщения в канал

```
1. (id, k, _) <- STORAGE[id]    // получение информации о канале
2. c = E(k, m)                  // шифрование сообщения AEAD алгоритмом
3. h = HMAC(id, c)              // хеш шифртекста
4. s = S(pk, h)                 // подпись хеша
5. p = P(h)                     // подтверждение сообщения доказательством работы
6. (id, pk, c, s, p) -> SERVER  // отправление сообщения на сервер 
```

## Получение сообщения из канала

```
1. (id, pk, c, s, p) <- SERVER  // получение сообщения от сервера
2. (id, k, RL) <- STORAGE[id]   // получение информации о канале
3. pk in? RL                    // проверка участника канала
4. ok =? V(pk, c, s)            // проверка подписи участника
5. ok =? V(c, p)                // проверка доказательства работы
6. m = D(k, c)                  // расшифрование сообщения
7. m -> STORAGE                 // сохранение сообщения
```
